**Goal:** Generate five 7.24 ER diagram SVGs (BPMN, DMN, CMMN, History, Identity) from `docs/documentation/reference/db-schema/schema.mermaid`, and wire them into the docs page that currently shows the 7.22 diagrams.

**Architecture:** A Python script parses `schema.mermaid` into entities/relationships/class-assignments, splits them into 5 fixed table groups, emits one `.mmd` file per group, and renders each with `@mermaid-js/mermaid-cli` (`mmdc`, invoked via `npx`). Output SVGs land in the same static image directory as the 7.22 originals, named `erd_724_*.svg`. `index.md` is updated to reference them.

**Tech Stack:** Python 3 (stdlib only — `re`, `subprocess`, `pathlib`, `tempfile`), `@mermaid-js/mermaid-cli` (already a transitive dependency via `mermaid` + `@docusaurus/theme-mermaid` in `package.json`; invoked via `npx -y @mermaid-js/mermaid-cli`, no install step needed).

## Global Constraints

- Source of truth: `docs/documentation/reference/db-schema/schema.mermaid` — do not hand-edit table/column data; always derive from this file.
- Do not modify or delete the 7.22 SVGs (`erd_722_*.svg`) or any other file referencing them outside `index.md`.

```python
# scripts/erd_schema.py
"""Parse schema.mermaid into entities, relationships, and class assignments."""
import re

ENTITY_RE = re.compile(r'^\s{4}(ACT_\w+)\s*\{(.*?)\n\s{4}\}', re.MULTILINE | re.DOTALL)
REL_RE = re.compile(
    r'^\s*(ACT_\w+)\s+([}|][o|]--[o|][|{])\s+(ACT_\w+)\s*:\s*"([^"]+)"',
    re.MULTILINE,
)
CLASS_RE = re.compile(r'^\s*class\s+(ACT_\w+)\s+(\w+)', re.MULTILINE)

TABLE_GROUPS = {
    "bpmn": [
        "ACT_GE_PROPERTY", "ACT_GE_BYTEARRAY", "ACT_GE_SCHEMA_LOG",
        "ACT_RE_DEPLOYMENT", "ACT_RE_PROCDEF", "ACT_RE_CAMFORMDEF",
        "ACT_RU_EXECUTION", "ACT_RU_JOB", "ACT_RU_JOBDEF", "ACT_RU_TASK",
        "ACT_RU_IDENTITYLINK", "ACT_RU_VARIABLE", "ACT_RU_EVENT_SUBSCR",
        "ACT_RU_INCIDENT", "ACT_RU_AUTHORIZATION", "ACT_RU_FILTER",
        "ACT_RU_METER_LOG", "ACT_RU_TASK_METER_LOG", "ACT_RU_EXT_TASK",
        "ACT_RU_BATCH",
    ],
    "dmn": ["ACT_RE_DECISION_DEF", "ACT_RE_DECISION_REQ_DEF"],
    "cmmn": ["ACT_RE_CASE_DEF", "ACT_RU_CASE_EXECUTION", "ACT_RU_CASE_SENTRY_PART"],
    "history": None,
    "identity": None,
}


def parse_schema(text):
    entities = {m.group(1): m.group(2) for m in ENTITY_RE.finditer(text)}
    rels = [
        (m.group(1), m.group(2), m.group(3), m.group(4))
        for m in REL_RE.finditer(text)
    ]
    classes = {m.group(1): m.group(2) for m in CLASS_RE.finditer(text)}
    return entities, rels, classes
```

```python
def resolve_groups(entities):
    groups = {k: v for k, v in TABLE_GROUPS.items() if v is not None}
    groups["history"] = sorted(n for n in entities if n.startswith("ACT_HI_"))
    groups["identity"] = sorted(n for n in entities if n.startswith("ACT_ID_"))

    grouped = [t for tables in groups.values() for t in tables]
    if len(grouped) != len(set(grouped)):
        seen, dupes = set(), set()
        for t in grouped:
            (dupes if t in seen else seen).add(t)
        raise ValueError(f"tables assigned to multiple groups: {sorted(dupes)}")

    missing = set(entities) - set(grouped)
    if missing:
        raise ValueError(f"tables not assigned to any group: {sorted(missing)}")

    return groups
```

```python
# scripts/erd_build.py
"""Build per-group mermaid ER sub-diagrams from parsed schema data."""


def extract_header(text):
    lines = text.splitlines()
    init_line = lines[0]
    classdef_lines = [l for l in lines if l.strip().startswith("classDef")]
    return init_line, classdef_lines


def build_subdiagram(tables, entities, rels, classes, init_line, classdef_lines):
    out = [init_line, "erDiagram"]
    out.extend("    " + l.strip() for l in classdef_lines)

    for t in tables:
        out.append(f"    {t} {{{entities[t]}\n    }}")

    for a, rel, b, col in rels:
        if a in tables and b in tables:
            out.append(f'    {a} {rel} {b} : "{col}"')

    for t in tables:
        if t in classes:
            out.append(f"    class {t} {classes[t]}")

    return "\n".join(out) + "\n"
```

### Task 4: Render pipeline + CLI entry point

**Interfaces:**
- Consumes: `parse_schema`, `resolve_groups` from `erd_schema`; `extract_header`, `build_subdiagram` from `erd_build`
- Produces: a runnable script `python3 scripts/generate_erd_diagrams.py` with no arguments, that writes `erd_724_{bpmn,dmn,cmmn,history,identity}.svg` into `static/img/documentation/user-guide/process-engine/database/` and prints one line per generated file. This task has no unit test (it shells out to `npx`/`mmdc`, which is an integration concern) — its own run *is* the test, verified manually in Step 3.

- [ ] **Step 1: Write the script**

```python
#!/usr/bin/env python3
"""Generate grouped 7.24 ER diagram SVGs from schema.mermaid.

Usage: python3 scripts/generate_erd_diagrams.py
Requires: npx (Node.js) available on PATH; @mermaid-js/mermaid-cli is
fetched on demand via `npx -y`, no separate install step needed.
"""
import subprocess
import sys
import tempfile
from pathlib import Path

sys.path.insert(0, str(Path(__file__).parent))
from erd_schema import parse_schema, resolve_groups
from erd_build import extract_header, build_subdiagram

REPO_ROOT = Path(__file__).parent.parent
SCHEMA_PATH = REPO_ROOT / "docs/documentation/reference/db-schema/schema.mermaid"
OUT_DIR = REPO_ROOT / "static/img/documentation/user-guide/process-engine/database"


def main():
    text = SCHEMA_PATH.read_text()
    entities, rels, classes = parse_schema(text)
    groups = resolve_groups(entities)
    init_line, classdef_lines = extract_header(text)

    OUT_DIR.mkdir(parents=True, exist_ok=True)

    with tempfile.TemporaryDirectory() as tmp:
        tmp_path = Path(tmp)
        for group, tables in groups.items():
            mmd_content = build_subdiagram(tables, entities, rels, classes, init_line, classdef_lines)
            mmd_path = tmp_path / f"{group}.mmd"
            mmd_path.write_text(mmd_content)

            svg_path = OUT_DIR / f"erd_724_{group}.svg"
            result = subprocess.run(
                [
                    "npx", "-y", "@mermaid-js/mermaid-cli",
                    "-i", str(mmd_path),
                    "-o", str(svg_path),
                    "-b", "transparent",
                ],
                capture_output=True,
                text=True,
            )
            if result.returncode != 0:
                print(f"FAILED rendering group '{group}':\n{result.stderr}", file=sys.stderr)
                sys.exit(1)
            print(f"Generated {svg_path} ({len(tables)} tables)")


if __name__ == "__main__":
    main()
```
